S -> S S | IF S | IF S ELSE S | variable = VALUE | COMMENT | FUNCTION ;
COMMENT -> # STRING | ''' STRING ''' ;
FUNCTION -> def variable ;
IF -> if BOOLEAN : ;
ELIF -> elif BOOLEAN : S | ELIF ELIF | e ;
ELSE -> else: S | e ;
VALUE -> ( VALUE ) | MATH | string ;
BOOLEAN -> True | False | VARVAL EVALUATOR VARVAL | ( BOOLEAN ) | BOOLEAN or BOOLEAN | BOOLEAN and BOOLEAN | not BOOLEAN ;
VARVAL -> variable | VALUE ;
EVALUATOR -> == | <= | < | > | >= | != ;
MATH -> angka | angka OPERATOR angka | ( MATH ) ;
OPERATOR -> + | - | * | / | mod ;
TRIPLEQUOTE -> ''';
STRING -> string | STRING STRING ;
