S -> S S | IF S | IF S ELSE S | VARIABEL = VALUE | # string | TRIPLEQUOTE string TRIPLEQUOTE ;
IF -> if CONDITION : ;
ELSE -> else : ;
VALUE -> MATH | string ;
CONDITION -> ( VALUE == VALUE ) | VALUE == VALUE ;
MATH -> angka | angka OPERATOR angka | ( angka ) ;
OPERATOR -> + | - | * | / | mod ;
TRIPLEQUOTE -> ''';
